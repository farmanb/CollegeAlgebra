  \def\xmin{-2}
  \def\xmax{5}
  \def\ymin{-2}
  \def\ymax{5}

  \def\a{1}
  \def\b{4}
  \def\c{4}
  \def\d{-1}
  
  \draw [very thick,->] (\xmin,0) -- (\xmax+.1,0) node [anchor=south] {$x$};
  \draw [very thick,->] (0,\ymin) -- (0,\ymax+.1) node [anchor=west] {$y$};
  
  \path [draw, help lines, opacity=.5]  (\xmin,\ymin) grid (\xmax,\ymax);

  \foreach \i in {-2,...,5}
  \draw [very thick] (\i,2.5pt) -- +(0,-5pt)
	[very thick] (2.5pt,\i) -- +(-5pt,0);
	
	\coordinate (P) at (\a,\b);
	\coordinate (Q) at (\c,\d);
	\coordinate (R) at (\a,\d);
	
	\draw [thick,blue] (P) -- (Q) -- (R) -- (P);


	\node at (P) {$\bullet$};
	\node at (Q) {$\bullet$};
	\node at (R) {$\bullet$};
	\node [above] at (P) {$(\a,\b)$};
	\node [right] at (Q) {$(\c,\d)$};
	\node [below] at (R) {$(\a,\d)$};
